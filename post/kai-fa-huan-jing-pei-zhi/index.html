REDIS

# 切换到安装目录准备下载安装包
cd /usr/local/src 

# 获取redis资源
wget http://download.redis.io/releases/redis-4.0.8.tar.gz

# 解压
tar xzvf redis-r.0.8.tar.gz

# 安装
cd redis-4.0.8

make
cd src
make install PREFIX=/usr/local/redis

# 移动配置文件到安装目录
mkdir /usr/local/redis/etc
cd ..
mv redis.conf /usr/local/redis/etc

# 修改配置 
vi /usr/local/redis/etc/redis.conf

# 后台运行
daemonize yes

# 日志文件
logfile:"/root/logs/redis.log"

# 远程连接
bind 0.0.0.0

# cp 执行命令到 bin
cp redis-benchmark /usr/bin/
cp redis-check-aof /usr/bin/
cp redis-check-rdb /usr/bin/
cp redis-cli /usr/bin/
cp redis-sentinel /usr/bin/
cp redis-server /usr/bin/

# 将redis加入到开机启动
vi /etc/rc.local

# 启动redis并指定配置文件
/usr/local/redis/bin/redis-server /usr/local/redis/etc/redis.conf

# 停止redis
pkill redis

RabbitMQ

# 安装必须插件
yum -y install make gcc gcc-c++ kernel-devel m4 ncurses-devel openssl-devel unixODBC-devel

# 安装 erlang
yum install erlang

# 进入 erlang
erl

# 推出 erlang
q().

# 安装 RabbitMQ
wget https://www.rabbitmq.com/releases/rabbitmq-server/v3.6.8/rabbitmq-server-3.6.8-1.el7.noarch.rpm

rpm --import https://www.rabbitmq.com/rabbitmq-release-signing-key.asc

yum install rabbitmq-server-3.6.8-1.el7.noarch.rpm

rpm -i --nodeps rabbitmq-server-3.6.8-1.el7.noarch.rpm

# 启动 RabbitMQ
/sbin/service rabbitmq-server start

# 创建 admin
rabbitmqctl add_user admin 123456

# 授权
rabbitmqctl  set_permissions -p "/" admin '.*' '.*' '.*'

# 赋予administrator角色
rabbitmqctl set_user_tags admin administrator

# 开启控制台
rabbitmq-plugins enable rabbitmq_management

# 访问
${IP}:15672

# 查看用户列表
rabbitmqctl list_users

# 修改密码
rabbitmqctl change_password ${username} '${new_password}'
